```typescript
export interface SentimentComparison {
  divergenceScore: number;
  winnerIndex: number; // 0 or 1
  trajectoryA: number[];
  trajectoryB: number[];
}

export function compareSentimentCurves(
  curveA: number[],
  curveB: number[]
): SentimentComparison {
  const len = Math.min(curveA.length, curveB.length);
  let sumDiff = 0;
  for (let i = 0; i < len; i++) {
    sumDiff += curveB[i] - curveA[i];
  }
  const avgDiff = sumDiff / len;
  return {
    divergenceScore: Math.abs(avgDiff),
    winnerIndex: avgDiff >= 0 ? 1 : 0,
    trajectoryA: curveA,
    trajectoryB: curveB,
  };
}

export interface KeywordOverlap {
  shared: string[];
  uniqueA: string[];
  uniqueB: string[];
}

export function computeKeywordOverlap(
  kwA: string[],
  kwB: string[]
): KeywordOverlap {
  const setB = new Set(kwB);
  const setA = new Set(kwA);
  return {
    shared: kwA.filter((k) => setB.has(k)),
    uniqueA: kwA.filter((k) => !setB.has(k)),
    uniqueB: kwB.filter((k) => !setA.has(k)),
  };
}
```

Create `backend/simulations/concordia/game_masters/nested_war_game_gm.py`:

```python
"""
WarGame Nested World Game Master.

Master GM coordinates multiple country sub-GMs. Each country has:
- Independent agent pool
- Language-specific prompts
- Local grounded variables (trust_index, belief_score)
- Cross-border information spillover via master GM

Architecture:
  MasterGM
    ├── SubGM(country="CN", lang="zh-Hans", agents=[...])
    ├── SubGM(country="US", lang="en", agents=[...])
    └── SubGM(country="JP", lang="ja", agents=[...])

Each tick:
  1. Master GM collects events from all sub-GMs
  2. Master GM determines cross-border spillover effects
  3. Master GM injects spillover events into affected sub-GMs
  4. Sub-GMs run one tick independently
  5. Master GM updates global grounded variables
"""
from .base_gm import BaseGameMaster

LANGUAGE_PROMPTS = {
    "en": "Respond in English.",
    "zh-Hans": "请用简体中文回答。",
    "zh-Hant": "請用繁體中文回答。",
    "ja": "日本語で回答してください。",
    "ko": "한국어로 답변해 주세요.",
}


class CountrySubGM:
    """Sub-GM managing one country's agent pool and local dynamics."""

    def __init__(self, country_id, language, agents, trust_index_init=50):
        self.country_id = country_id
        self.language = language
        self.agents = agents
        self.trust_index = trust_index_init
        self.belief_score = 50
        self.local_events = []
        self.language_instruction = LANGUAGE_PROMPTS.get(language, LANGUAGE_PROMPTS["en"])

    def run_tick(self, external_events, llm_client, tick):
        """Run one simulation tick for this country."""
        context_parts = [
            f"Country: {self.country_id} | Language: {self.language}",
            f"Current trust index: {self.trust_index}/100",
            f"Current belief score: {self.belief_score}/100",
        ]
        if external_events:
            context_parts.append(f"External events: {external_events}")

        tick_events = []
        for agent in self.agents:
            prompt = (
                f"{self.language_instruction}\n"
                f"You are {agent['name']}, a citizen of {self.country_id}.\n"
                f"Persona: {agent.get('persona', '')}\n"
                f"Context: {'; '.join(context_parts)}\n"
                f"Tick {tick}: What is your reaction to the current situation?"
            )
            response = llm_client.generate(prompt)
            tick_events.append({
                "agent_id": agent["id"],
                "agent_name": agent["name"],
                "country": self.country_id,
                "content": response,
                "tick": tick,
            })

        self.local_events.extend(tick_events)
        return tick_events

    def update_variables(self, trust_delta, belief_delta):
        self.trust_index = max(0, min(100, self.trust_index + trust_delta))
        self.belief_score = max(0, min(100, self.belief_score + belief_delta))


class NestedWarGameGM(BaseGameMaster):
    """Master GM orchestrating nested country sub-GMs."""

    def __init__(self, config, llm_client):
        super().__init__(config, llm_client)
        self.sub_gms = {}
        self.information_penetration_rate = 0
        self._init_countries(config)

    def _init_countries(self, config):
        nested = config.get("nested_config", {})
        countries = nested.get("countries", [])
        agents_by_country = {}

        for agent in config.get("agents", []):
            country = agent.get("country", "default")
            agents_by_country.setdefault(country, []).append(agent)

        for country_cfg in countries:
            cid = country_cfg["id"]
            self.sub_gms[cid] = CountrySubGM(
                country_id=cid,
                language=country_cfg.get("language", "en"),
                agents=agents_by_country.get(cid, []),
                trust_index_init=country_cfg.get("trust_index_init", 50),
            )

    def get_initial_grounded_vars(self):
        variables = {"information_penetration_rate": 0}
        for cid, sub in self.sub_gms.items():
            variables[f"{cid}_trust_index"] = sub.trust_index
            variables[f"{cid}_belief_score"] = sub.belief_score
        return variables

    def run_tick(self, tick, context):
        all_events = []
        # 1. Each country runs independently
        for cid, sub in self.sub_gms.items():
            spillover = self._get_spillover_for(cid, tick)
            events = sub.run_tick(spillover, self.llm_client, tick)
            all_events.extend(events)

        # 2. Master determines cross-border effects
        self._compute_spillover(all_events, tick)

        # 3. Update grounded variables
        grounded = {"information_penetration_rate": self.information_penetration_rate}
        for cid, sub in self.sub_gms.items():
            grounded[f"{cid}_trust_index"] = sub.trust_index
            grounded[f"{cid}_belief_score"] = sub.belief_score

        return all_events, grounded

    def _get_spillover_for(self, country_id, tick):
        """Collect events from other countries that spill over."""
        spillover = []
        for cid, sub in self.sub_gms.items():
            if cid != country_id and sub.local_events:
                recent = [e for e in sub.local_events if e["tick"] == tick - 1]
                if recent:
                    spillover.append(f"News from {cid}: {recent[0]['content'][:200]}")
        return spillover

    def _compute_spillover(self, events, tick):
        """Use LLM to assess cross-border impact and update variables."""
        if not events:
            return

        summary = "; ".join(
            f"[{e['country']}] {e['agent_name']}: {e['content'][:100]}"
            for e in events[:10]
        )
        prompt = (
            f"Analyze these geopolitical reactions from tick {tick}:\n{summary}\n\n"
            "For each country, estimate:\n"
            "- trust_delta (-10 to +10): change in public trust\n"
            "- belief_delta (-10 to +10): change in belief/cognitive shift\n"
            "- info_penetration_delta (0 to 5): misinformation spread increase\n"
            "Return JSON: {\"countries\": {\"XX\": {\"trust_delta\": N, \"belief_delta\": N}}, "
            "\"info_penetration_delta\": N}"
        )
        try:
            response = self.llm_client.generate(prompt)
            import json
            data = json.loads(response)
            for cid, deltas in data.get("countries", {}).items():
                if cid in self.sub_gms:
                    self.sub_gms[cid].update_variables(
                        deltas.get("trust_delta", 0),
                        deltas.get("belief_delta", 0),
                    )
            self.information_penetration_rate = max(0, min(100,
                self.information_penetration_rate + data.get("info_penetration_delta", 0)))
        except Exception:
            pass  # Fail gracefully on parse errors

    def is_complete(self, tick, grounded_vars):
        """Complete after max ticks or if any trust index hits extreme."""
        max_ticks = self.config.get("max_ticks", 20)
        if tick >= max_ticks:
            return True
        for cid in self.sub_gms:
            trust = grounded_vars.get(f"{cid}_trust_index", 50)
            if trust <= 5 or trust >= 95:
                return True
        return False

    def get_branches(self):
        return [
            {"label": "Diplomatic", "description": "Pursue diplomatic negotiations and de-escalation"},
            {"label": "Military", "description": "Military posturing and deterrence strategy"},
            {"label": "Economic", "description": "Economic sanctions and trade leverage"},
        ]
```

const WORKER_COUNT = 4;

async function embedChunksParallel(chunks: { id: string; content: string }[]): Promise<Map<string, number[]>> {
  const results = new Map<string, number[]>();

  // Split chunks across workers
  const perWorker = Math.ceil(chunks.length / WORKER_COUNT);
  const workers: Worker[] = [];
  const promises: Promise<void>[] = [];

  for (let w = 0; w < WORKER_COUNT; w++) {
    const workerChunks = chunks.slice(w * perWorker, (w + 1) * perWorker);
    if (workerChunks.length === 0) continue;

    const worker = new Worker(new URL("../workers/embeddingWorker.ts", import.meta.url));
    workers.push(worker);

    promises.push(
      new Promise<void>((resolve) => {
        worker.onmessage = (evt) => {
          const { embeddings, batchIndex } = evt.data;
          for (let i = 0; i < embeddings.length; i++) {
            results.set(workerChunks[i].id, embeddings[i]);
          }
          worker.terminate();
          resolve();
        };
      })
    );

    worker.postMessage({
      type: "embed_batch",
      texts: workerChunks.map((c) => c.content),
      batchIndex: w,
    });
  }

  await Promise.all(promises);
  return results;
}

/**
 * HNSW Recall@10 Benchmark
 *
 * Measures recall by comparing HNSW approximate results against
 * exact brute-force results for random query vectors.
 *
 * Usage: cd backend && bun run scripts/benchmark-hnsw.ts
 */
import { query } from "../src/db/client";

const NUM_QUERIES = 20;

async function benchmark() {
  // Get sample vectors from documents table
  const samples = await query(
    `SELECT id, embedding FROM documents WHERE embedding IS NOT NULL ORDER BY random() LIMIT $1`,
    [NUM_QUERIES]
  );

  if (samples.rows.length === 0) {
    console.log("No documents with embeddings found. Skipping benchmark.");
    return;
  }

  let totalRecall = 0;

  for (const sample of samples.rows) {
    const vec = sample.embedding;

    // Exact nearest neighbors (brute force)
    const exact = await query(
      `SELECT id FROM documents WHERE embedding IS NOT NULL
       ORDER BY embedding <=> $1::vector LIMIT 10`,
      [vec]
    );

    // HNSW approximate (uses index)
    const approx = await query(
      `SET LOCAL hnsw.ef_search = 100;
       SELECT id FROM documents WHERE embedding IS NOT NULL
       ORDER BY embedding <=> $1::vector LIMIT 10`,
      [vec]
    );

    const exactIds = new Set(exact.rows.map((r: any) => r.id));
    const approxIds = approx.rows.map((r: any) => r.id);
    const hits = approxIds.filter((id: string) => exactIds.has(id)).length;
    totalRecall += hits / 10;
  }

  const avgRecall = totalRecall / samples.rows.length;
  console.log(`\nHNSW Recall@10: ${(avgRecall * 100).toFixed(1)}% (target: ≥85%)`);
  console.log(`Queries tested: ${samples.rows.length}`);

  if (avgRecall >= 0.85) {
    console.log("✓ PASS: Recall meets target");
  } else {
    console.log("✗ FAIL: Recall below target — consider increasing m or ef_search");
  }

  process.exit(0);
}

benchmark().catch(console.error);

    .filter((e: any) => e.event_type === "grounded_vars_update")

            <Tooltip />

      </div>

        <h3 className="text-lg font-semibold mb-4">
          {t("warGame.countryBreakdown", "Country Breakdown")}
        </h3>
        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
          {countries.map((c: string) => {
            const latest = varSnapshots[varSnapshots.length - 1] || {};
            const trust = latest[`${c}_trust_index`] ?? 50;
            const belief = latest[`${c}_belief_score`] ?? 50;
            return (
              <div key={c} className="border rounded-lg p-4">
                <h4 className="font-semibold text-lg">{c}</h4>

**Step 3: Create SDK client wrapper**

Create `sdk/src/index.ts`:

```typescript
export type { paths, components } from "../generated/api-types";

export interface ParaVerseClientConfig {
  baseURL: string;
  token?: string;
}

export function createParaVerseClient(config: ParaVerseClientConfig) {
  const headers = (): Record<string, string> => {
    const h: Record<string, string> = { "Content-Type": "application/json" };
    if (config.token) h["Authorization"] = `Bearer ${config.token}`;
    return h;
  };

  async function request<T>(method: string, path: string, body?: unknown): Promise<T> {
    const res = await fetch(`${config.baseURL}${path}`, {
      method,
      headers: headers(),
      body: body ? JSON.stringify(body) : undefined,
    });
    const json = await res.json();
    if (!res.ok) throw new Error(json.error || `HTTP ${res.status}`);
    return json.data;
  }

  return {
    // Auth
    login: (email: string, password: string) =>
      request<{ accessToken: string; refreshToken: string }>("POST", "/api/v1/auth/login", { email, password }),

    // Projects
    listProjects: () => request<any[]>("GET", "/api/v1/projects"),
    createProject: (data: { name: string; description?: string; scenario_type: string }) =>
      request<any>("POST", "/api/v1/projects", data),

    // Simulations
    createSimulation: (data: { project_id: string; config: Record<string, unknown> }) =>
      request<any>("POST", "/api/v1/simulations", data),
    startSimulation: (id: string) =>
      request<any>("POST", `/api/v1/simulations/${id}/start`),
    getSimulationStatus: (id: string) =>
      request<any>("GET", `/api/v1/simulations/${id}/status`),
    getSimulationEvents: (id: string, params?: { limit?: number; cursor?: string }) => {
      const qs = new URLSearchParams();
      if (params?.limit) qs.set("limit", String(params.limit));
      if (params?.cursor) qs.set("cursor", params.cursor);
      return request<any[]>("GET", `/api/v1/simulations/${id}/events?${qs}`);
    },

    // Reports
    generateReport: (simId: string) =>
      request<any>("POST", `/api/v1/simulations/${simId}/report`),
    getReport: (simId: string) =>
      request<any>("GET", `/api/v1/simulations/${simId}/report`),

    // Graph
    buildGraph: (projectId: string) =>
      request<any>("POST", `/api/v1/projects/${projectId}/graph/build`),
    getGraph: (projectId: string) =>
      request<any>("GET", `/api/v1/projects/${projectId}/graph`),

    // Tasks
    getTask: (taskId: string) =>
      request<any>("GET", `/api/v1/tasks/${taskId}`),

    // Interview
    interviewAgent: (simId: string, agentId: string, question: string) =>
      request<any>("POST", `/api/v1/simulations/${simId}/interview`, { agent_id: agentId, question }),
  };
}
```
