```typescript
import { query } from "../db/client";
import type { AcceptanceMatrix, StakeholderAcceptance } from "@shared/types/simulation";
import { logger } from "../utils/logger";

export async function computeAcceptanceMatrix(
  simulationId: string
): Promise<AcceptanceMatrix> {
  // Fetch agent_action events that have group and branch metadata
  const result = await query<{
    agent_id: string;
    content: string;
    metadata: string;
  }>(
    `SELECT agent_id, content, metadata
     FROM simulation_events
     WHERE simulation_id = $1
       AND event_type = 'agent_action'
       AND metadata::text LIKE '%group%'
     ORDER BY sim_timestamp`,
    [simulationId]
  );

  const groupBranchMap = new Map<string, { positive: number; negative: number; neutral: number; total: number }>();
  const groups = new Set<string>();
  const branches = new Set<string>();

  for (const row of result.rows) {
    const meta = typeof row.metadata === "string" ? JSON.parse(row.metadata) : row.metadata;
    const group = meta.group || "unknown";
    const branch = meta.branch || "default";
    groups.add(group);
    branches.add(branch);

    const key = `${group}::${branch}`;
    if (!groupBranchMap.has(key)) {
      groupBranchMap.set(key, { positive: 0, negative: 0, neutral: 0, total: 0 });
    }
    const counts = groupBranchMap.get(key)!;
    counts.total++;

    // Simple sentiment classification based on content keywords
    const lower = (row.content || "").toLowerCase();
    if (lower.includes("support") || lower.includes("agree") || lower.includes("approve") || lower.includes("favor")) {
      counts.positive++;
    } else if (lower.includes("oppose") || lower.includes("reject") || lower.includes("against") || lower.includes("hurt")) {
      counts.negative++;
    } else {
      counts.neutral++;
    }
  }

  const cells: StakeholderAcceptance[] = [];
  for (const [key, counts] of groupBranchMap) {
    const [group, branch] = key.split("::");
    const acceptance_score = counts.total > 0
      ? Math.round(((counts.positive + counts.neutral * 0.5) / counts.total) * 100)
      : 50;
    const sentiment: StakeholderAcceptance["sentiment"] =
      acceptance_score >= 60 ? "positive" : acceptance_score >= 40 ? "neutral" : "negative";

    cells.push({
      stakeholder_group: group,
      branch_label: branch,
      acceptance_score,
      sentiment,
      sample_size: counts.total,
    });
  }

  return {
    simulation_id: simulationId,
    groups: [...groups],
    branches: [...branches],
    cells,
    generated_at: new Date().toISOString(),
  };
}
```

```tsx
import { useQuery } from "@tanstack/react-query";
import { matrixApi } from "../../api/matrix";
import type { AcceptanceMatrix as MatrixType } from "@shared/types/simulation";

interface Props {
  simulationId: string;
}

const SENTIMENT_COLORS = {
  positive: "bg-green-100 text-green-800 border-green-300",
  neutral: "bg-yellow-50 text-yellow-800 border-yellow-300",
  negative: "bg-red-100 text-red-800 border-red-300",
} as const;

function getCellColor(score: number): string {
  if (score >= 70) return "bg-green-500 text-white";
  if (score >= 55) return "bg-green-300 text-green-900";
  if (score >= 45) return "bg-yellow-200 text-yellow-900";
  if (score >= 30) return "bg-orange-300 text-orange-900";
  return "bg-red-500 text-white";
}

export function AcceptanceMatrix({ simulationId }: Props) {
  const { data: matrix, isLoading, error } = useQuery({
    queryKey: ["acceptance-matrix", simulationId],
    queryFn: () => matrixApi.getAcceptanceMatrix(simulationId),
    refetchInterval: 10_000,
  });

  if (isLoading) return <div className="animate-pulse h-48 bg-gray-100 rounded" />;
  if (error || !matrix) return <div className="text-red-500 text-sm">Failed to load acceptance matrix</div>;
  if (matrix.cells.length === 0) return <div className="text-gray-400 text-sm">No acceptance data available yet</div>;

  return (
    <div className="overflow-x-auto">
      <h3 className="text-lg font-semibold mb-3">Stakeholder Acceptance Matrix</h3>
      <table className="min-w-full border-collapse">
        <thead>
          <tr>
            <th className="border px-3 py-2 bg-gray-50 text-left text-sm font-medium">Group / Branch</th>
            {matrix.branches.map((branch) => (
              <th key={branch} className="border px-3 py-2 bg-gray-50 text-center text-sm font-medium">
                {branch.replace("Branch ", "").replace(": ", "\n")}
              </th>
            ))}
          </tr>
        </thead>
        <tbody>
          {matrix.groups.map((group) => (
            <tr key={group}>
              <td className="border px-3 py-2 text-sm font-medium capitalize">{group.replace(/_/g, " ")}</td>
              {matrix.branches.map((branch) => {
                const cell = matrix.cells.find(
                  (c) => c.stakeholder_group === group && c.branch_label === branch
                );
                if (!cell) return <td key={branch} className="border px-3 py-2 text-center text-gray-300">—</td>;
                return (
                  <td key={branch} className={`border px-3 py-2 text-center text-sm font-bold ${getCellColor(cell.acceptance_score)}`}>
                    {cell.acceptance_score}%
                    <div className="text-xs font-normal opacity-75">n={cell.sample_size}</div>
                  </td>
                );
              })}
            </tr>
          ))}
        </tbody>
      </table>
      <p className="text-xs text-gray-400 mt-2">Generated: {new Date(matrix.generated_at).toLocaleString()}</p>
    </div>
  );
}
```

```tsx
import { LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer } from "recharts";
import type { SimEvent } from "@shared/types/simulation";

interface Props {
  events: SimEvent[];
}

interface TimePoint {
  tick: number;
  public_approval: number;
  policy_support: number;
  opposition_strength: number;
  media_coverage: number;
}

export function PolicyImpactChart({ events }: Props) {
  const groundedVarEvents = events.filter((e) => e.event_type === "grounded_var");

  // Group by tick and build timeline
  const tickMap = new Map<number, TimePoint>();
  for (const evt of groundedVarEvents) {
    const tick = evt.sim_timestamp ?? 0;
    if (!tickMap.has(tick)) {
      tickMap.set(tick, { tick, public_approval: 0, policy_support: 0, opposition_strength: 0, media_coverage: 0 });
    }
    const point = tickMap.get(tick)!;
    const meta = typeof evt.metadata === "string" ? JSON.parse(evt.metadata) : evt.metadata;
    const name = meta?.name || evt.content;
    const value = meta?.value ?? 0;
    if (name === "public_approval") point.public_approval = value;
    if (name === "policy_support") point.policy_support = value;
    if (name === "opposition_strength") point.opposition_strength = value;
    if (name === "media_coverage") point.media_coverage = value;
  }

  const data = [...tickMap.values()].sort((a, b) => a.tick - b.tick);

  if (data.length === 0) return null;

  return (
    <div>
      <h3 className="text-lg font-semibold mb-3">Policy Impact Timeline</h3>
      <ResponsiveContainer width="100%" height={300}>
        <LineChart data={data}>
          <CartesianGrid strokeDasharray="3 3" />
          <XAxis dataKey="tick" label={{ value: "Tick", position: "bottom" }} />
          <YAxis domain={[0, 100]} />
          <Tooltip />
          <Legend />
          <Line type="monotone" dataKey="public_approval" stroke="#22c55e" name="Public Approval" strokeWidth={2} />
          <Line type="monotone" dataKey="policy_support" stroke="#3b82f6" name="Policy Support" strokeWidth={2} />
          <Line type="monotone" dataKey="opposition_strength" stroke="#ef4444" name="Opposition" strokeWidth={2} />
          <Line type="monotone" dataKey="media_coverage" stroke="#f59e0b" name="Media Coverage" strokeWidth={1} strokeDasharray="5 5" />
        </LineChart>
      </ResponsiveContainer>
    </div>
  );
}
```

```typescript
import { Hono } from "hono";
import { HTTPException } from "hono/http-exception";
import { authMiddleware, type AuthContext } from "../middleware/auth";
import { getSimulationForOwner } from "../db/queries/simulations";
import type { ApiResponse } from "@shared/types/api";
import { readdir, stat, access } from "node:fs/promises";
import { join } from "node:path";
import { logger } from "../utils/logger";

const checkpoint = new Hono();
checkpoint.use("*", authMiddleware);

const CHECKPOINT_DIR = process.env.CHECKPOINT_DIR || "/tmp/paraverse_checkpoints";

interface CheckpointInfo {
  filename: string;
  tick: number | null;
  created_at: string;
  size_bytes: number;
}

// List checkpoints for a simulation
checkpoint.get("/:simulationId/checkpoints", async (c) => {
  const auth = c.get("auth") as AuthContext;
  const simulationId = c.req.param("simulationId");

  const sim = await getSimulationForOwner(simulationId, auth.userId);
  if (!sim) throw new HTTPException(404, { message: "Simulation not found" });

  const simCheckpointDir = join(CHECKPOINT_DIR, simulationId);
  const checkpoints: CheckpointInfo[] = [];

  try {
    await access(simCheckpointDir);
    const files = await readdir(simCheckpointDir);
    for (const file of files) {
      if (!file.endsWith(".pkl")) continue;
      const filePath = join(simCheckpointDir, file);
      const fileStat = await stat(filePath);
      const tickMatch = file.match(/tick(\d+)/);
      checkpoints.push({
        filename: file,
        tick: tickMatch ? parseInt(tickMatch[1]) : null,
        created_at: new Date(fileStat.mtimeMs).toISOString(),
        size_bytes: fileStat.size,
      });
    }
  } catch {
    // Directory doesn't exist = no checkpoints
  }

  checkpoints.sort((a, b) => (b.tick ?? 0) - (a.tick ?? 0));

  return c.json({
    success: true,
    data: { checkpoints },
    error: null,
  } satisfies ApiResponse);
});

// Load a checkpoint (sends IPC command to running simulation)
checkpoint.post("/:simulationId/checkpoints/load", async (c) => {
  const auth = c.get("auth") as AuthContext;
  const simulationId = c.req.param("simulationId");
  const { filename } = await c.req.json<{ filename: string }>();

  const sim = await getSimulationForOwner(simulationId, auth.userId);
  if (!sim) throw new HTTPException(404, { message: "Simulation not found" });
  if (sim.engine !== "concordia") {
    throw new HTTPException(400, { message: "Checkpoints only supported for Concordia engine" });
  }

  const filePath = join(CHECKPOINT_DIR, simulationId, filename);
  try {
    await access(filePath);
  } catch {
    throw new HTTPException(404, { message: "Checkpoint file not found" });
  }

  // Import simulation service to send IPC command
  const { getSimulationService } = await import("../services/simulationService");
  const simService = getSimulationService();
  await simService.sendCommand(simulationId, {
    type: "load_checkpoint",
    checkpoint_path: filePath,
  });

  return c.json({
    success: true,
    data: { loaded: filename },
    error: null,
  } satisfies ApiResponse);
});

export { checkpoint };
```

```tsx
import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
import { checkpointApi } from "../../api/simulation";

interface CheckpointInfo {
  filename: string;
  tick: number | null;
  created_at: string;
  size_bytes: number;
}

interface Props {
  simulationId: string;
  disabled?: boolean;
}

export function CheckpointManager({ simulationId, disabled }: Props) {
  const queryClient = useQueryClient();

  const { data, isLoading } = useQuery({
    queryKey: ["checkpoints", simulationId],
    queryFn: () => checkpointApi.list(simulationId),
    refetchInterval: 15_000,
  });

  const saveMutation = useMutation({
    mutationFn: () => checkpointApi.save(simulationId),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["checkpoints", simulationId] }),
  });

  const loadMutation = useMutation({
    mutationFn: (filename: string) => checkpointApi.load(simulationId, filename),
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ["simulation-status", simulationId] }),
  });

  const checkpoints: CheckpointInfo[] = data?.checkpoints || [];

  return (
    <div className="border rounded-lg p-4 bg-white">
      <div className="flex items-center justify-between mb-3">
        <h3 className="text-sm font-semibold">Checkpoints</h3>
        <button
          onClick={() => saveMutation.mutate()}
          disabled={disabled || saveMutation.isPending}
          className="px-3 py-1 text-xs bg-blue-600 text-white rounded hover:bg-blue-700 disabled:opacity-50"
        >
          {saveMutation.isPending ? "Saving..." : "Save Checkpoint"}
        </button>
      </div>

      {isLoading ? (
        <div className="text-sm text-gray-400">Loading checkpoints...</div>
      ) : checkpoints.length === 0 ? (
        <div className="text-sm text-gray-400">No checkpoints saved yet</div>
      ) : (
        <ul className="space-y-2 max-h-48 overflow-y-auto">
          {checkpoints.map((cp) => (
            <li key={cp.filename} className="flex items-center justify-between text-sm border-b pb-2">
              <div>
                <span className="font-mono text-xs">{cp.tick !== null ? `Tick ${cp.tick}` : cp.filename}</span>
                <span className="text-gray-400 ml-2 text-xs">
                  {new Date(cp.created_at).toLocaleTimeString()}
                </span>
                <span className="text-gray-400 ml-1 text-xs">
                  ({Math.round(cp.size_bytes / 1024)}KB)
                </span>
              </div>
              <button
                onClick={() => loadMutation.mutate(cp.filename)}
                disabled={loadMutation.isPending}
                className="px-2 py-0.5 text-xs border rounded hover:bg-gray-100 disabled:opacity-50"
              >
                {loadMutation.isPending ? "Loading..." : "Load"}
              </button>
            </li>
          ))}
        </ul>
      )}

  );
}
```

```sql
-- Documents table: searchable content
ALTER TABLE documents ADD COLUMN IF NOT EXISTS search_vector tsvector;
UPDATE documents SET search_vector = to_tsvector('english', coalesce(title, '') || ' ' || coalesce(content, ''));
CREATE INDEX IF NOT EXISTS idx_documents_search ON documents USING GIN (search_vector);

-- Create trigger to auto-update tsvector on insert/update
CREATE OR REPLACE FUNCTION documents_search_trigger() RETURNS trigger AS $$
BEGIN
  NEW.search_vector := to_tsvector('english', coalesce(NEW.title, '') || ' ' || coalesce(NEW.content, ''));
  RETURN NEW;
END
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trig_documents_search ON documents;
CREATE TRIGGER trig_documents_search
  BEFORE INSERT OR UPDATE ON documents
  FOR EACH ROW EXECUTE FUNCTION documents_search_trigger();

-- Simulation events: searchable content
ALTER TABLE simulation_events ADD COLUMN IF NOT EXISTS search_vector tsvector;

-- Note: For partitioned tables, index must be created on each partition
-- The trigger will handle new inserts
CREATE OR REPLACE FUNCTION sim_events_search_trigger() RETURNS trigger AS $$
BEGIN
  NEW.search_vector := to_tsvector('english', coalesce(NEW.content, ''));
  RETURN NEW;
END
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trig_sim_events_search ON simulation_events;
CREATE TRIGGER trig_sim_events_search
  BEFORE INSERT OR UPDATE ON simulation_events
  FOR EACH ROW EXECUTE FUNCTION sim_events_search_trigger();

-- GIN indexes on partitions
DO $$
DECLARE
  part TEXT;
BEGIN
  FOR part IN
    SELECT inhrelid::regclass::text
    FROM pg_inherits
    WHERE inhparent = 'simulation_events'::regclass
  LOOP
    EXECUTE format('CREATE INDEX IF NOT EXISTS idx_%s_search ON %s USING GIN (search_vector)', replace(part, '.', '_'), part);
  END LOOP;
END $$;

-- Ontology nodes: searchable name + description
ALTER TABLE ontology_nodes ADD COLUMN IF NOT EXISTS search_vector tsvector;
UPDATE ontology_nodes SET search_vector = to_tsvector('english', coalesce(label, '') || ' ' || coalesce(properties::text, ''));
CREATE INDEX IF NOT EXISTS idx_ontology_search ON ontology_nodes USING GIN (search_vector);

CREATE OR REPLACE FUNCTION ontology_search_trigger() RETURNS trigger AS $$
BEGIN
  NEW.search_vector := to_tsvector('english', coalesce(NEW.label, '') || ' ' || coalesce(NEW.properties::text, ''));
  RETURN NEW;
END
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trig_ontology_search ON ontology_nodes;
CREATE TRIGGER trig_ontology_search
  BEFORE INSERT OR UPDATE ON ontology_nodes
```

```typescript
import { query } from "../db/client";
import { getVectorService } from "./vectorService";
import { logger } from "../utils/logger";

interface HybridSearchParams {
  query: string;
  projectId: string;
  table: "documents" | "simulation_events" | "ontology_nodes";
  limit?: number;
  mode?: "hybrid" | "vector" | "text";
  vectorWeight?: number; // 0-1, default 0.6
}

interface SearchResult {
  id: string;
  title?: string;
  content: string;
  score: number;
  source: "vector" | "text" | "hybrid";
}

const TABLE_CONFIG = {
  documents: {
    idCol: "id",
    contentCol: "content",
    titleCol: "title",
    embeddingCol: "embedding",
    projectFilter: "project_id",
  },
  simulation_events: {
    idCol: "id",
    contentCol: "content",
    titleCol: null,
    embeddingCol: "embedding",
    projectFilter: null, // filtered by simulation_id separately
  },
  ontology_nodes: {
    idCol: "id",
    contentCol: "label",
    titleCol: null,
    embeddingCol: "embedding",
    projectFilter: "project_id",
  },
} as const;

export async function hybridSearch(params: HybridSearchParams): Promise<SearchResult[]> {
  const {
    query: searchQuery,
    projectId,
    table,
    limit = 10,
    mode = "hybrid",
    vectorWeight = 0.6,
  } = params;

  const config = TABLE_CONFIG[table];
  const textWeight = 1 - vectorWeight;

  if (mode === "text") {
    return textSearch(searchQuery, table, config, projectId, limit);
  }

  if (mode === "vector") {
    return vectorSearch(searchQuery, table, config, projectId, limit);
  }

  // Hybrid: run both and fuse results using Reciprocal Rank Fusion (RRF)
  const [textResults, vectorResults] = await Promise.all([
    textSearch(searchQuery, table, config, projectId, limit * 2),
    vectorSearch(searchQuery, table, config, projectId, limit * 2),
  ]);

  return reciprocalRankFusion(textResults, vectorResults, textWeight, vectorWeight, limit);
}

async function textSearch(
  searchQuery: string,
  table: string,
  config: (typeof TABLE_CONFIG)[keyof typeof TABLE_CONFIG],
  projectId: string,
  limit: number
): Promise<SearchResult[]> {
  const tsQuery = searchQuery
    .split(/\s+/)
    .filter(Boolean)
    .map((w) => `${w}:*`)
    .join(" & ");

  const projectClause = config.projectFilter ? `AND ${config.projectFilter} = $2` : "";
  const params = config.projectFilter ? [tsQuery, projectId, limit] : [tsQuery, limit];
  const limitParam = config.projectFilter ? "$3" : "$2";

  const titleSelect = config.titleCol ? `, ${config.titleCol} as title` : "";

  const result = await query<{ id: string; content: string; title?: string; rank: number }>(
    `SELECT ${config.idCol} as id, ${config.contentCol} as content${titleSelect},
            ts_rank(search_vector, to_tsquery('english', $1)) as rank
     FROM ${table}
     WHERE search_vector @@ to_tsquery('english', $1) ${projectClause}
     ORDER BY rank DESC
     LIMIT ${limitParam}`,
    params
  );

  return result.rows.map((r) => ({
    id: r.id,
    title: r.title,
    content: r.content,
    score: r.rank,
    source: "text" as const,
  }));
}

async function vectorSearch(
  searchQuery: string,
  table: string,
  config: (typeof TABLE_CONFIG)[keyof typeof TABLE_CONFIG],
  projectId: string,
  limit: number
): Promise<SearchResult[]> {
  const vectorService = getVectorService();
  const embedding = await vectorService.embed(searchQuery);
  const embeddingStr = `[${embedding.join(",")}]`;

  const projectClause = config.projectFilter ? `AND ${config.projectFilter} = $2` : "";
  const params = config.projectFilter
    ? [embeddingStr, projectId, limit]
    : [embeddingStr, limit];
  const limitParam = config.projectFilter ? "$3" : "$2";

  const titleSelect = config.titleCol ? `, ${config.titleCol} as title` : "";

  const result = await query<{ id: string; content: string; title?: string; distance: number }>(
    `SELECT ${config.idCol} as id, ${config.contentCol} as content${titleSelect},
            ${config.embeddingCol} <=> $1::vector as distance
     FROM ${table}
     WHERE ${config.embeddingCol} IS NOT NULL ${projectClause}
     ORDER BY distance ASC
     LIMIT ${limitParam}`,
    params
  );

  return result.rows.map((r) => ({
    id: r.id,
    title: r.title,
    content: r.content,
    score: 1 - r.distance, // Convert distance to similarity
    source: "vector" as const,
  }));
}

function reciprocalRankFusion(
  textResults: SearchResult[],
  vectorResults: SearchResult[],
  textWeight: number,
  vectorWeight: number,
  limit: number
): SearchResult[] {
  const k = 60; // RRF constant
  const scores = new Map<string, { result: SearchResult; score: number }>();

  textResults.forEach((r, i) => {
    const rrfScore = textWeight / (k + i + 1);
    const existing = scores.get(r.id);
    if (existing) {
      existing.score += rrfScore;
    } else {
      scores.set(r.id, { result: { ...r, source: "hybrid" }, score: rrfScore });
    }
  });

  vectorResults.forEach((r, i) => {
    const rrfScore = vectorWeight / (k + i + 1);
    const existing = scores.get(r.id);
    if (existing) {
      existing.score += rrfScore;
    } else {
      scores.set(r.id, { result: { ...r, source: "hybrid" }, score: rrfScore });
    }
  });

  return [...scores.values()]
    .sort((a, b) => b.score - a.score)
    .slice(0, limit)
    .map((s) => ({ ...s.result, score: s.score }));
}
```

```sql
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  resource_type TEXT NOT NULL, -- 'simulation', 'api_call', 'export'

  metadata JSONB DEFAULT '{}'
);

CREATE INDEX idx_usage_user_type ON usage_records (user_id, resource_type);
CREATE INDEX idx_usage_consumed ON usage_records (consumed_at);

-- Monthly usage view for quick quota checks
CREATE OR REPLACE VIEW monthly_usage AS
SELECT
  user_id,
  resource_type,
  date_trunc('month', consumed_at) AS month,
  count(*) AS usage_count
FROM usage_records
GROUP BY user_id, resource_type, date_trunc('month', consumed_at);
```

```typescript
import { query } from "../db/client";

export async function getMonthlyUsageCount(
  userId: string,
  resourceType: string
): Promise<number> {
  const result = await query<{ count: string }>(
    `SELECT count(*)::text as count FROM usage_records
     WHERE user_id = $1
       AND resource_type = $2
       AND consumed_at >= date_trunc('month', now())`,
    [userId, resourceType]
  );
  return parseInt(result.rows[0]?.count || "0");
}

export async function recordUsage(
  userId: string,
  resourceType: string,
  simulationId?: string,
  metadata?: Record<string, unknown>
): Promise<void> {
  await query(
    `INSERT INTO usage_records (user_id, resource_type, simulation_id, metadata)
     VALUES ($1, $2, $3, $4)`,
    [userId, resourceType, simulationId ?? null, JSON.stringify(metadata ?? {})]
  );
}
```

**Step 4: Write quota enforcement middleware**

Create `backend/src/middleware/quotaEnforcement.ts`:

```typescript
import type { Context, Next } from "hono";
import { HTTPException } from "hono/http-exception";
import { getUserById } from "../db/queries/users";
import { getMonthlyUsageCount } from "../db/queries/usage";
import { logger } from "../utils/logger";

interface UserQuota {
  simulations_per_month: number;
  max_agents: number;
}

const QUOTA_FIELD_MAP: Record<string, keyof UserQuota> = {
  simulation: "simulations_per_month",
};

export function quotaCheck(resourceType: string) {
  return async (c: Context, next: Next) => {
    const auth = c.get("auth") as { userId: string } | undefined;
    if (!auth?.userId) {
      throw new HTTPException(401, { message: "Authentication required" });
    }

    const quotaField = QUOTA_FIELD_MAP[resourceType];
    if (!quotaField) {
      // No quota defined for this resource type, allow
      return next();
    }

    try {
      const user = await getUserById(auth.userId);
      if (!user) throw new HTTPException(401, { message: "User not found" });

      const quota = (user.quota as UserQuota) || { simulations_per_month: 2, max_agents: 50 };
      const maxAllowed = quota[quotaField];
      const currentUsage = await getMonthlyUsageCount(auth.userId, resourceType);

      if (currentUsage >= maxAllowed) {
        logger.warn({ userId: auth.userId, resourceType, currentUsage, maxAllowed }, "Quota exceeded");
        throw new HTTPException(429, {
          message: `Monthly ${resourceType} quota exceeded (${currentUsage}/${maxAllowed}). Upgrade your plan for more.`,
        });
      }
    } catch (err) {
      if (err instanceof HTTPException) throw err;
      // Fail open on quota check errors (don't block users due to DB issues)
      logger.error({ err, userId: auth.userId }, "Quota check failed, allowing request");
    }

    return next();
  };
}
```
